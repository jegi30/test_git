# Creation du depot local
git init

# Ajout dans le sac des fichiers
git add *.c
git add *.h

# on archive le sac dans le depot local
git commit -m "first commit"

# on renomme la branche principale du depot local en main
git branch -M main

# on connecte le depot github créé via le nom "origin"
git remote add origin https://github.com/jegi30/test_git.git

# on archive un etat du logicel
git push -u origin main

# Modification du code existant par un editeur par exemple doc.txt
# ici on rajoute un fichier recap.txt
# puis on mets à jour l'ensemble
git add recap.txt
git add doc.txt
git commit -m "Ajout recap.txt"
git push -u origin main

# pour recuperer un projet github
# on cree un repertoire vide, on positionne la le BASH GIT dans ce repertoire.
git clone https://github.com/jegi30/test_git.git

# mettre à jour son repertoire local avec le depot distant
git pull origin main
